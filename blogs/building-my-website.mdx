## Example Code

```jsx
function LikeButton() {
  const [likes, setLikes] = useState(0);
  return (
    <button onClick={() => setLikes(likes + 1)}>
      👍 {likes} Like{likes !== 1 ? 's' : ''}
    </button>
  );
}
```
